def create_run_folder(base_dir="runs_ibm_hardware"):
    os.makedirs(base_dir, exist_ok=True)
    existing = [d for d in os.listdir(base_dir) if d.startswith("run_")]

    if not existing:
        run_id = 1
    else:
        nums = [int(d.split("_")[1]) for d in existing]
        run_id = max(nums) + 1

    run_path = os.path.join(base_dir, f"run_{run_id}")
    os.makedirs(run_path)
    return run_path, run_id

class DigitsDataset(Dataset):
    def __init__(self, csv_file, label=0):
        self.df = pd.read_csv(csv_file, header=None)
        self.df = self.df[self.df.iloc[:, -1] == label]

    def __len__(self):
        return len(self.df)

    def __getitem__(self, idx):
        image = self.df.iloc[idx, :-1].values.astype(np.float32)
        image = image / 16.0
        image = (image - 0.5) * 2
        image = torch.tensor(image).view(1, 8, 8)
        return image, 0

def setup_ibm_backend():
    """Setup IBM Quantum backend"""
    if USE_REAL_HARDWARE:
        try:
            service = QiskitRuntimeService()
            backend = service.backend(IBM_BACKEND_NAME)
            log(f"✅ Connected to real hardware: {backend.name}")

            log(f"   Qubits: {backend.num_qubits}")
            
            # Configure options for hardware
            options = Options()
            options.execution.shots = 4096  # More shots for better accuracy
            options.resilience_level = 2   # Error mitigation
            options.dynamical_decoupling.enable = True
            options.dynamical_decoupling.sequence_type = "XY4"
            
            return backend, options, service
        except Exception as e:
            log(f"⚠️ Error connecting to hardware: {e}")
            log("Falling back to noisy simulator")
            return setup_fake_backend()
    else:
        return setup_fake_backend()

class Critic(nn.Module):
    def __init__(self):
        super().__init__()
        self.model = nn.Sequential(
            nn.Flatten(),
            nn.Linear(64, 64),
            nn.LeakyReLU(0.2),
            nn.Linear(64, 32),
            nn.LeakyReLU(0.2),
            nn.Linear(32, 1)
        )

    def forward(self, x):
        return self.model(x)

@qml.qnode(dev, interface="torch", diff_method="parameter-shift")
def quantum_circuit(noise, weights):
    """Optimized circuit for IBM hardware"""
    weights = weights.reshape(q_depth, n_qubits)

    # Simple encoding (no amplitude embedding for hardware)
    norm = torch.norm(noise)
    if norm.item() == 0:
        norm = torch.tensor(1.0)
    
    # Angle encoding instead of amplitude (hardware-efficient)
    for i in range(n_qubits):
        qml.RY(noise[i] * np.pi, wires=i)

    for i in range(q_depth):
        for y in range(n_qubits):
            qml.RY(weights[i][y], wires=y)
        # Nearest-neighbor CNOTs (matches IBM topology)
        for y in range(n_qubits - 1):
            qml.CNOT(wires=[y, y + 1])

    return qml.probs(wires=range(measured_qubits))

def partial_measure(noise, weights):
    probs = quantum_circuit(noise, weights)
    # Ensure correct size
    if len(probs) > patch_size:
        probs = probs[:patch_size]
    elif len(probs) < patch_size:
        padding = torch.zeros(patch_size - len(probs), device=probs.device)
        probs = torch.cat([probs, padding])
    probs = probs / (torch.sum(probs) + 1e-8)
    return probs

class PatchQuantumGenerator(nn.Module):
    def __init__(self, n_generators):
        super().__init__()

        self.q_params = nn.ParameterList([
            nn.Parameter(torch.rand(q_depth * n_qubits) * 0.1)
            for _ in range(n_generators)
        ])

        self.fc = nn.Sequential(
            nn.Linear(IMAGE_PIXELS, 128),
            nn.ReLU(),
            nn.Linear(128, IMAGE_PIXELS),
            nn.ReLU()
        )

        self.conv = nn.Sequential(
            nn.ConvTranspose2d(1, 8, 3, padding=1),
            nn.ReLU(),
            nn.ConvTranspose2d(8, 1, 3, padding=1),
            nn.Tanh()
        )

    def forward(self, x):
        images = torch.empty(x.size(0), 0).to(device)

        for params in self.q_params:
            patches = torch.empty(0, patch_size).to(device)

            for elem in x:
                q_out = partial_measure(elem, params)
                q_out = q_out.float().unsqueeze(0)
                patches = torch.cat((patches, q_out), dim=0)

            # Pad or trim patches
            if patches.size(1) < patch_size:
                pad = patch_size - patches.size(1)
                patches = torch.cat([patches, torch.zeros(patches.size(0), pad).to(device)], dim=1)
            else:
                patches = patches[:, :patch_size]

            images = torch.cat((images, patches), dim=1)

        if images.size(1) < IMAGE_PIXELS:
            pad = IMAGE_PIXELS - images.size(1)
            images = torch.cat([images, torch.zeros(images.size(0), pad).to(device)], dim=1)
        else:
            images = images[:, :IMAGE_PIXELS]
        
        x = self.fc(images)
        x = x.view(-1, 1, 8, 8)
        x = self.conv(x)

        return x.view(-1, 64)

# ================== GRADIENT PENALTY ==================
def gradient_penalty(critic, real, fake):
    alpha = torch.rand(real.size(0), 1).to(device)
    interpolated = alpha * real + (1 - alpha) * fake
    interpolated.requires_grad_(True)

    prob = critic(interpolated)

    gradients = torch.autograd.grad(
        outputs=prob,
        inputs=interpolated,
        grad_outputs=torch.ones_like(prob),
        create_graph=True
    )[0]

    gradients = gradients.view(gradients.size(0), -1)
    return ((gradients.norm(2, dim=1) - 1) ** 2).mean()

def calculate_fid_simple(real, fake):
    """Simplified FID for 8x8 images"""
    real_flat = real.view(real.size(0), -1).numpy()
    fake_flat = fake.view(fake.size(0), -1).numpy()
    
    mu1, sigma1 = real_flat.mean(axis=0), np.cov(real_flat, rowvar=False)
    mu2, sigma2 = fake_flat.mean(axis=0), np.cov(fake_flat, rowvar=False)
    
    # Add regularization
    sigma1 += np.eye(sigma1.shape[0]) * 1e-6
    sigma2 += np.eye(sigma2.shape[0]) * 1e-6
    
    diff = mu1 - mu2
    covmean = sqrtm(sigma1 @ sigma2)
    if np.iscomplexobj(covmean):
        covmean = covmean.real
    
    fid = diff @ diff + np.trace(sigma1 + sigma2 - 2 * covmean)
    return float(fid) if np.isfinite(fid) else 0.0

def calculate_ssim_simple(real, fake):
    real_np = real.numpy()
    fake_np = fake.numpy()
    scores = []
    for i in range(len(real_np)):
        r = real_np[i].squeeze()
        f = fake_np[i].squeeze()
        try:
            scores.append(ssim(r, f, data_range=2.0))
        except:
            scores.append(0.0)
    return float(np.mean(scores))

def calculate_diversity_simple(fake):
    fake = fake.view(fake.size(0), -1)
    distances = []
    for i in range(min(100, len(fake))):
        for j in range(i + 1, min(100, len(fake))):
            distances.append(torch.norm(fake[i] - fake[j]).item())
    return float(np.mean(distances)) if distances else 0.0

def generate_fake_images(total_samples):
    chunks = []
    with torch.no_grad():
        for start in range(0, total_samples, BATCH_SIZE):

            noise = torch.randn(current_batch, n_qubits).to(device)
            chunk = generator(noise).view(current_batch, 1, 8, 8).cpu().detach()
            chunks.append(chunk)
            if total_samples >= BATCH_SIZE * 2:
                log(f"Generated eval fake samples {start + current_batch}/{total_samples}")
    return torch.cat(chunks, dim=0)
